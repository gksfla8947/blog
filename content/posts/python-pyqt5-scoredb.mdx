이름, 나이, 점수가 담겨있는 데이터베이스를 불러와 Show, Add, Delete, Find, Increase를 할 수 있는 GUI를 만들어본다.

## 기능 소개

- **Add** : name, age, score을 입력받고 그 정보를 DB에 추가
- **Del** : name을 입력받아 해당 정보를 DB에서 삭제
- **Find** : name을 입력받아 해당 이름의 정보를 찾아 Result에 출력
- **Inc** : name과 amount를 입력받아 name에 해당하는 score을 amount만큼 증가
- **Show** : key를 기준으로 오름차순으로 DB를 Result에 출력

## 코드

```python
import pickle
import sys
from PyQt5.QtWidgets import (QWidget, QPushButton,
                             QHBoxLayout, QVBoxLayout, QApplication, QLabel,
                             QComboBox, QLineEdit, QGridLayout, QTextBrowser)


class ScoreDB(QWidget):

    def __init__(self):
        super().__init__()
        self.dbfilename = 'assignment6.dat'
        self.scoredb = []
        self.readScoreDB()
        self.initUI()
        self.showScoreDB('Name')

        self.commandInfo = ""
        self.nameInfo = None
        self.ageInfo = None
        self.scoreInfo = None
        self.amountInfo = None
        self.keyInfo = "Name"

    def initUI(self):

        self.name = QLabel('Name: ')
        self.age = QLabel('Age: ')
        self.score = QLabel('Score: ')
        self.amount = QLabel('Amount: ')
        self.key = QLabel('Key: ')
        self.result = QLabel('Result: ')

        self.keyBox = QComboBox()
        self.keyBox.addItem('Name')
        self.keyBox.addItem('Age')
        self.keyBox.addItem('Score')
        self.keyBox.currentIndexChanged.connect(self.appendKeyInfo)

        self.nameEdit = QLineEdit()
        self.nameEdit.textChanged.connect(self.appendNameInfo)
        self.ageEdit = QLineEdit()
        self.ageEdit.textChanged.connect(self.appendAgeInfo)
        self.scoreEdit = QLineEdit()
        self.scoreEdit.textChanged.connect(self.appendScoreInfo)
        self.amountEdit = QLineEdit()
        self.amountEdit.textChanged.connect(self.appendAmountInfo)
        self.resultEdit = QTextBrowser()

        self.addButton = QPushButton('Add')
        self.addButton.clicked.connect(self.buttonClicked)

        self.delButton = QPushButton('Del')
        self.delButton.clicked.connect(self.buttonClicked)

        self.findButton = QPushButton('Find')
        self.findButton.clicked.connect(self.buttonClicked)

        self.incButton = QPushButton('Inc')
        self.incButton.clicked.connect(self.buttonClicked)

        self.showButton = QPushButton('Show')
        self.showButton.clicked.connect(self.buttonClicked)

        hbox1 = QHBoxLayout()
        hbox1.addWidget(self.addButton)
        hbox1.addWidget(self.delButton)
        hbox1.addWidget(self.findButton)
        hbox1.addWidget(self.incButton)
        hbox1.addWidget(self.showButton)

        grid1 = QGridLayout()
        grid2 = QGridLayout()

        grid1.addWidget(self.name, 1, 0)
        grid1.addWidget(self.nameEdit, 1, 1)
        grid1.addWidget(self.age, 1, 2)
        grid1.addWidget(self.ageEdit, 1, 3)
        grid1.addWidget(self.score, 1, 4)
        grid1.addWidget(self.scoreEdit, 1, 5)
        grid1.addWidget(self.amount, 2, 2)
        grid1.addWidget(self.amountEdit, 2, 3)
        grid1.addWidget(self.key, 2, 4)
        grid1.addWidget(self.keyBox, 2, 5)

        grid2.addWidget(self.result, 1, 0)
        grid2.addWidget(self.resultEdit, 2, 0, 5, 1)

        vbox = QVBoxLayout()
        vbox.addLayout(grid1)
        vbox.addLayout(hbox1)
        vbox.addLayout(grid2)
        vbox.addStretch(1)

        self.setLayout(vbox)

        self.setGeometry(300, 300, 500, 250)
        self.setWindowTitle('ScoreDB')
        self.show()

    # def closeEvent(self, event):
    #    self.writeScoreDB()

    def closeEvent(self, event):
        self.writeScoreDB()

    def readScoreDB(self):
        try:
            fH = open(self.dbfilename, 'rb')
        except FileNotFoundError as e:
            self.scoredb = []
            return
        self.scoredb = pickle.load(fH)
        fH.close()

    # write the data into person db
    def writeScoreDB(self):
        fH = open(self.dbfilename, 'wb')
        pickle.dump(self.scoredb, fH)
        fH.close()

    def showScoreDB(self, keyname):
        self.resultEdit.clear()
        for p in sorted(self.scoredb, key=lambda person: person[keyname]):
            temp = ""
            for attr in sorted(p):
                temp += attr + "=" + str(p[attr]) + "\t\t"
            self.resultEdit.append(temp)

    def addScoreDB(self, name, age, score):
        try:
            record = {'Name': name, 'Age': int(age), 'Score': int(score)}
            self.scoredb += [record]
            self.resultEdit.clear()
            self.showScoreDB(self.keyInfo)
            self.clearEdit()
            self.initValue()
        except TypeError or ValueError:
            self.resultEdit.clear()
            self.resultEdit.setText("Enter the correct information(Name, Age, Score)")
            self.clearEdit()
            self.initValue()

    def delScoreDB(self, name):
        for p in self.scoredb:
            if p['Name'] == name:
                self.scoredb.remove(p)
                self.resultEdit.clear()
                self.showScoreDB(self.keyInfo)
        self.clearEdit()
        self.initValue()

    def findScoreDB(self, name):
        temp = ""
        for p in self.scoredb:
            if p['Name'] == name:
                self.resultEdit.clear()
                for attr in sorted(p):
                    temp += attr + "=" + str(p[attr]) + "\t\t"
                self.resultEdit.setText(temp)
        self.initValue()
        self.clearEdit()

    def incScoreDB(self, name, amount):
        try:
            for p in self.scoredb:
                if p['Name'] == name:
                    for key in p:
                        if key == 'Score':
                            newValue = int(p[key]) + int(amount)
                            p[key] = newValue
            self.resultEdit.clear()
            self.showScoreDB(self.keyInfo)
            self.initValue()
            self.clearEdit()
        except TypeError:
            self.resultEdit.clear()
            self.resultEdit.setText("Enter the Amount")
            self.clearEdit()
            self.initValue()

    def buttonClicked(self):
        sender = self.sender()

        if sender.text() == "Show":
            self.showScoreDB(self.keyInfo)
        elif sender.text() == "Add":
            self.addScoreDB(self.nameInfo, self.ageInfo, self.scoreInfo)
        elif sender.text() == "Del":
            self.delScoreDB(self.nameInfo)
        elif sender.text() == "Find":
            self.findScoreDB(self.nameInfo)
        elif sender.text() == "Inc":
            self.incScoreDB(self.nameInfo, self.amountInfo)

    def appendNameInfo(self):
        sender = self.sender()
        if (self.isNone(sender.text())):
            self.nameInfo = None
        else:
            self.nameInfo = sender.text()

    def appendAgeInfo(self):
        sender = self.sender()
        if (self.isNone(sender.text())):
            self.ageInfo = None
        else:
            self.ageInfo = sender.text()

    def appendScoreInfo(self):
        sender = self.sender()
        if (self.isNone(sender.text())):
            self.scoreInfo = None
        else:
            self.scoreInfo = sender.text()

    def appendAmountInfo(self):
        sender = self.sender()
        if (self.isNone(sender.text())):
            self.amountInfo = None
        else:
            self.amountInfo = sender.text()

    def appendKeyInfo(self):
        sender = self.sender()
        self.keyInfo = sender.currentText()

    def clearEdit(self):
        self.nameEdit.setText(None)
        self.ageEdit.setText(None)
        self.scoreEdit.setText(None)
        self.amountEdit.setText(None)

    def initValue(self):
        self.nameInfo = None
        self.ageInfo = None
        self.scoreInfo = None
        self.amountInfo = None

    def isNone(self, nullText):
        if nullText == "":
            return True


if __name__ == '__main__':
    app = QApplication(sys.argv)
    ex = ScoreDB()
    sys.exit(app.exec_())
```

과제 첫 수행 당시 가장 어려웠던 점은 레이아웃 배치였다. QH,QVBox 레이아웃에 대한 이해가 부족한 상황에서 첫줄은 위젯 6개, 두번째 줄은 4개, 세번째 줄은 5개 등 각 줄마다 위젯 수가 다르니 비율이 맞지 않는 문제가 있었다. 하지만 grid레이아웃을 통해 name, age, score, amount, keyname값을 입력 받는 부분은 간격을 통일해 배치하는 것에 성공했고 QHBox레이 아웃을 통해 다섯 개의 버튼을 가운데에 일정 비율로 꽉차게 배치했다. 그리고 grid레이아웃을 하나 더 생성하여 result창을 배치하였고 최종적으로 vbox라는 QVBox레이아웃을 생성하여 세개의 레이아웃을 수직으로 일정간격으로 배치하는데 성공하였다.
